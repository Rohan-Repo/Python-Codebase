Since we had to add year because of our mistake we had to do makemigrations and migrate again
(.venv) C:\MyCodes\Python\Django\Django-Movies-App>python manage.py sqlmigrate movies 0002
BEGIN;
--
-- Add field year to movie
--
CREATE TABLE "new__movies_movie" ("id" integer NOT NULL PRIMARY KEY AUTOINCREMENT, "year" integer NOT NULL, "title" varchar(100) NOT NULL); 
INSERT INTO "new__movies_movie" ("id", "title", "year") SELECT "id", "title", 0 FROM "movies_movie";
DROP TABLE "movies_movie";
ALTER TABLE "new__movies_movie" RENAME TO "movies_movie";
COMMIT;
